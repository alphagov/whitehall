# check how many impacted rows where slug will be overwritten with slug_override
  SELECT count(*)
  FROM editions e
  WHERE e.slug_override IS NOT NULL
  AND e.slug_override != ''
  AND e.slug != e.slug_override
  AND e.state NOT IN ('deleted', 'superseded');
  => 1694

# number of nulls
  SELECT count(*)
  FROM editions e
  WHERE e.slug_override IS NOT NULL
  AND e.slug != e.slug_override
  AND e.state NOT IN ('deleted', 'superseded');
 => 4618

# number of blanks
  SELECT count(*)
  FROM editions e
  WHERE e.slug_override = ''
  AND e.slug != e.slug_override
  AND e.state NOT IN ('deleted', 'superseded');
=> 2924

# earlist where slug_override is blank
  SELECT min(updated_at)
  FROM editions e
  WHERE e.slug_override = ''
    AND e.slug != e.slug_override
    AND e.state NOT IN ('deleted', 'superseded');
=> 2026-04-09 14:10:06

 #number of post-publiction impacted
   SELECT count(*)
   FROM editions e
   WHERE state IN ('published','withdrawn', 'unpublished')
   AND e.slug_override IS NOT NULL
   AND e.slug_override != ''
   AND e.slug != e.slug_override;
=> 1647

#number of pre-publiction impacted
  SELECT count(*)
  FROM editions e
  WHERE state IN ('draft', 'submitted', 'rejected', 'scheduled')
  AND e.slug_override IS NOT NULL
  AND e.slug_override != ''
  AND e.slug != e.slug_override;
=> 47

#number of landing pages impacted
  SELECT id, state, slug, slug_from_title, slug_override
  FROM editions e
  WHERE e.slug_override IS NOT NULL
  AND e.slug_override != ''
  AND e.slug != e.slug_override
  AND e.type = 'PlanForChangeLandingPage'
  AND e.state NOT IN ('deleted', 'superseded');
=> 2

#number of non-english primaries with an SO
  SELECT count(*)
  FROM editions e
  WHERE e.primary_locale != 'en'
  AND e.slug_override IS NOT NULL
  AND e.slug_override != '';
=> 0

#number of non-english primaries where slug doesn't match SO
  SELECT count(*)
  FROM editions e
  WHERE e.primary_locale != 'en'
  AND e.slug_override IS NOT NULL
  AND e.slug_override != ''
  AND e.slug != e.slug_override;
=> 0

#slugs that use document id's and SO is different
  SELECT count(*)
  FROM editions e
    WHERE e.slug_override IS NOT NULL
    AND e.slug_override != ''
    AND slug != slug_override
    AND slug = CAST(document_id AS CHAR) COLLATE utf8mb4_unicode_ci;
 => 0
